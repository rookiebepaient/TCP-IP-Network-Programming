# send和recv函数
```C++
#include <sys/socekt.h>
/// @param [in] sockfd 表示与数据传输的连接的套接字件描述符
/// @param [in] buf 保存待传输数据的缓冲地址值
/// @param [in] nbytes  待传输的字节数
/// @param [in] flags   传输数据时指定的可
/// @return 成功返回发送的字节数，失败时返回-1
ssize_t send(int sockfd, const oid *buf, size_t nbytes, int falgs)

/// @param [in] sockfd 表示与数据传输的连接的套接字件描述符
/// @param [in] buf 保存接受数据的缓冲地址值
/// @param [in] nbytes  可接收最大的字节数
/// @param [in] flags   接收数据时指定的可
/// @return 成功返回接受的字节数，失败时返回-1
ssize_t send(int sockfd, const oid *buf, size_t nbytes, int falgs)
```
